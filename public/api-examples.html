POST Ham Transaction Gönder

/rpc/sendRawTx
JavaScript
Python
const crypto = require('crypto');

const timestamp = Date.now();
const txData = `from${to}${amount}${timestamp}`;
const txId = crypto.createHash('sha256').update(txData).digest('hex');

const tx = {
  tx_id: `TX_${txId}`,
  from_wallet: 'LT_sender',
  to_wallet: 'LT_receiver',
  type: 'TRANSFER',
  amount: 10 * 100000000,  // 10 LT
  fee: 100000,             // 0.001 LT
  timestamp: timestamp,
  payload: { priority: 'STANDARD' },
  signatures: [],
  sender_public_key: 'your_public_key',
  sender_signature: 'signature'
};

const response = await fetch(`${API_URL}/rpc/sendRawTx`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify(tx)
});

const result = await response.json();
console.log('TX ID:', result.tx_id);
import hashlib
import time

    'timestamp': timestamp,

Mesajları Alma

/chain (filtrele)
JavaScript
Python
const myWallet = 'LT_my_wallet';

const response = await fetch(`${API_URL}/chain`);
const { chain } = await response.json();

const messages = [];
for (const block of chain) {
  for (const tx of block.transactions) {
    if (tx.type === 'MESSAGE_PAYMENT' && tx.to_wallet === myWallet) {
      messages.push({
        from: tx.from_wallet,
        message: tx.payload.message,
        time: new Date(tx.timestamp),
        block: block.index
      });
    }
  }
}

console.log(`Toplam ${messages.length} mesaj`);
my_wallet = 'LT_my_wallet'

response = requests.get(f'{API_URL}/chain')
chain = response.json()['chain']

messages = []
for block in chain:
    for tx in block['transactions']:
        if tx['type'] == 'MESSAGE_PAYMENT' and tx['to_wallet'] == my_wallet:
            messages.append({
                'from': tx['from_wallet'],
                'message': tx['payload']['message'],
                'time': tx['timestamp'],
                'block': block['index']
            })

print(f"Toplam {len(messages)} mesaj")
