const data = await response.json();

Para Transferi (Yeni!)

POST
/rpc/transfer
Otomatik ücret hesaplamalı transfer - İmza gerektirir
JavaScript
Python
// İlk olarak: İşlem imzala (KeyManager ile)
const txData = {
  from_wallet: 'TRN1a2b3c...',
  to_wallet: 'TRN4d5e6f...',
  amount: 50 * 100000000, // 50 LT
  priority: 'MEDIUM' // Opsiyonel
};

// İmza oluştur (client-side)
const signature = signTransactionData(txData, privateKey);

// Transfer gönder
const response = await fetch('http://localhost:3000/rpc/transfer', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    ...txData,
    sender_public_key: publicKey,
    sender_signature: signature
  })
});

const data = await response.json();
console.log('TX ID:', data.tx_id);
console.log('Fee:', data.fee_readable);
tx_data = {
    'from_wallet': 'TRN1a2b3c...',
    'to_wallet': 'TRN4d5e6f...',
    'amount': 50 * 100000000,  # 50 LT
    'priority': 'MEDIUM'
}

# İmza oluştur
signature = sign_transaction(tx_data, private_key)

# Transfer gönder
response = requests.post('http://localhost:3000/rpc/transfer', json={
    **tx_data,
    'sender_public_key': public_key,
    'sender_signature': signature
})

data = response.json()
print(f"TX ID: {data['tx_id']}")
print(f"Fee: {data['fee_readable']}")
💡 Ücret Kademeleri:
  • Temel Ücret (Alıcının gelen transfer sayısına göre):
    • 0-49 transfer: 0.01%
    • 50-99 transfer: 0.025%
    • 100-199 transfer: 0.05%
    • 200+ transfer: 0.10%
  • Öncelik Ücreti (İşlem hızı):
    • STANDARD: +0%
    • LOW: +0.20%
    • MEDIUM: +0.60%
    • HIGH: +1%
